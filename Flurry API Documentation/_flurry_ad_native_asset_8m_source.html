68 + (kAssetType) nativeAssetTypeStringToEnum:(NSString*)strVal
69 {
70  if ((strVal == nil) || ([strVal isEqualToString:@""]))
71  {
72  return ASSET_TYPE_UNKNOWN;
73  }
74 
75  NSUInteger n = [[self nativeAssetTypeArrayInstance] indexOfObject:strVal];
76  if(n == NSNotFound) n = ASSET_TYPE_UNKNOWN;

78 }
